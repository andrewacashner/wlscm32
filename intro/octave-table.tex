    \pitchnames
    \octaveprimes \pitch{C}{\theoctave} 
    & \octavenumbers \pitch{C}{\theoctave}\tabularnewline
}
\stepcounter{octave}
\ifnum\value{octave} < 8
\repeat

\begin{tabular}{ll}
    \pitchnames
\end{tabular}
